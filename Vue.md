安装npm

~~~
npm install -g npm
~~~

安装vue-cli

~~~
npm install vue-cli -g   //全局安装 vue-cli
~~~

查看vue-cli版本

~~~
vue -V
~~~

卸载

~~~
npm uninstall vue-cli -g
~~~

设置镜像

~~~
npm config set registry https://registry.npm.taobao.org
~~~





创建项目

~~~
vue init webpack frontend
~~~

下载项目依赖

~~~
npm install
~~~

打包

~~~
npm run build
~~~



运行项目

~~~
npm run dev
~~~
